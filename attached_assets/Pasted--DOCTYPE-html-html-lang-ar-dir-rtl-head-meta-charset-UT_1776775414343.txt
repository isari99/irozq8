    <script>
        let direction = 1; // 1 للأمام، -1 للخلف
        let turn = 0;

        // توليد أوراق اللاعب عند البدء
        window.onload = () => {
const colors = ['red', 'blue', 'green', 'yellow'];
            const container = document.getElementById('my-cards');
            for(let i=0; i<7; i++) {
                const color = colors[Math.floor(Math.random()colors.length)];
                const val = Math.floor(Math.random()9);
                createCard(color, val);
            }
        };

        function createCard(color, val) {
            const card = document.createElement('div');
            card.className = card ${color};
            card.innerText = val;
            card.onclick = () => playThisCard(card);

            document.getElementById('my-cards').appendChild(card);
        }

        // أنيميشن سحب الورقة (طيران من المنتصف لليد)
        function animateCardDraw() {
            const deck = document.getElementById('main-deck');
            const hand = document.getElementById('my-cards');
            const deckRect = deck.getBoundingClientRect();
const fly = document.createElement('div');
            fly.className = 'flying-card';
            fly.style.left = deckRect.left + 'px';
            fly.style.top = deckRect.top + 'px';
            document.body.appendChild(fly);

            setTimeout(() => {
                const handRect = hand.getBoundingClientRect();
                fly.style.left = (window.innerWidth / 2) + 'px';
                fly.style.top = (window.innerHeight - 100) + 'px';
                fly.style.transform = 'rotate(360deg) scale(0.2)';
                fly.style.opacity = '0';
            }, 50);

            setTimeout(() => {
                fly.remove();
                createCard('yellow', '9');
            }, 750);
        }

        // لعب الورقة ومعالجة Skip و Reverse
        function playThisCard(el) {
            const val = el.innerText;
            const color = el.classList[1];

            // نقل الورقة للمنتصف
            document.getElementById('current-discard').innerHTML = <div class="card ${color}">${val}</div>;
            el.remove();

            // منطق القفز والعكس
            if (val == 'S'  val == 'Skip') {
                notify("تخطي الدور!");
                skipLogic();
            } else if (val == 'R' 
 val == 'Reverse') {
                notify("عكس الاتجاه!");
                reverseLogic();
            } else {
                normalNextTurn();
            }
function notify(msg) {
            const n = document.getElementById('game-notifier');
            n.innerText = msg; n.style.opacity = '1';
            setTimeout(() => n.style.opacity = '0', 1200);
        }

        function normalNextTurn() {
            turn = (turn + direction + 4) % 4;
            console.log("الدور الآن على: " + turn);
        }

        function skipLogic() {
            // تخطي: نتحرك خطوتين
            turn = (turn + (direction * 2) + 4) % 4;
            console.log("تم التخطي! الدور الآن على: " + turn);
        }

        function reverseLogic() {
            direction *= -1;
            normalNextTurn();
        }
